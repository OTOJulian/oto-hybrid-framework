```js
function resolveKey({ env = process.env, keyfileDir } = {}) { /* D-15 table exactly, per behavior above */ }
function buildSpawnCommand() { /* per behavior above */ }
function main() {
  const key = resolveKey();
  if (!key) { process.stderr.write('oto-exa-mcp: no Exa API key (EXA_API_KEY or ~/.oto/exa_api_key) — run /oto-settings-integrations\n'); process.exit(1); }
  const { cmd, args } = buildSpawnCommand();
  const child = require('node:child_process').spawn(cmd, args, { stdio: 'inherit', env: { ...process.env, EXA_API_KEY: key } });
  for (const sig of ['SIGTERM', 'SIGINT']) process.on(sig, () => child.kill(sig));
  child.on('error', () => { process.stderr.write('oto-exa-mcp: failed to spawn npx\n'); process.exit(1); });
  child.on('exit', (code, signal) => process.exit(signal ? 1 : (code ?? 1)));
}
module.exports = { resolveKey, buildSpawnCommand };
if (require.main === module) main();
```
