    const test = require('node:test');

    const REPO_ROOT = path.resolve(__dirname, '..');
    const OTO_TOOLS = path.join(REPO_ROOT, 'oto/bin/lib/oto-tools.cjs');

    function runInit(key) {
      const res = spawnSync(process.execPath, [OTO_TOOLS, 'init', key], {
        cwd: REPO_ROOT,
        encoding: 'utf8',
        env: { ...process.env, OTO_WORKSTREAM: '' }, // neutralize active workstream
      });
      assert.equal(res.status, 0, `init ${key} failed: stderr=${res.stderr}`);
      return JSON.parse(res.stdout);
    }

    test('phase 07 - workspace init handlers smoke', { concurrency: false }, async (t) => {
      await t.test('init new-workspace returns documented keys', () => {
        const out = runInit('new-workspace');
        // Per CONTEXT.md D-08: default_workspace_base, child_repos, worktree_available,
        // is_git_repo, cwd_repo_name, project_root
        for (const key of ['default_workspace_base', 'child_repos', 'worktree_available', 'is_git_repo', 'cwd_repo_name', 'project_root']) {
          assert.ok(key in out, `missing key: ${key} in ${JSON.stringify(out)}`);
        }
      });

      await t.test('init list-workspaces returns documented keys', () => {
        const out = runInit('list-workspaces');
        // Read oto/workflows/list-workspaces.md "Parse JSON for" line during read_first
        // and list those keys here. At minimum, expect a `workspaces` array.
        assert.ok('workspaces' in out, `missing 'workspaces' key in ${JSON.stringify(out)}`);

      });

      await t.test('init remove-workspace returns documented keys', () => {
        const out = runInit('remove-workspace');
