// bin/lib/install.cjs — orchestrator
'use strict';
const fs = require('node:fs');
const fsp = require('node:fs/promises');
const os = require('node:os');
const path = require('node:path');
const { resolveConfigDir } = require('./args.cjs');
const { copyTree, sha256File, removeTree, walkTree } = require('./copy-files.cjs');
const {
  injectMarkerBlock, removeMarkerBlock, OPEN_MARKER, CLOSE_MARKER,
} = require('./marker.cjs');
const { readState, writeState } = require('./install-state.cjs');
const { detectPresentRuntimes } = require('./runtime-detect.cjs');
const { version: OTO_VERSION } = require('../../package.json');

const SRC_KEYS = ['commands', 'agents', 'skills', 'hooks'];
const TRANSFORM_KEY = {
  commands: 'transformCommand',
  agents:   'transformAgent',
  skills:   'transformSkill',
  hooks:    'transformSkill', // hooks share skill transform shape at Phase 3
};

// Containment guard — refuses any path that, after resolution, escapes configDir
function assertWithin(configDir, relPath) {
  const joined = path.join(configDir, relPath);
  const resolvedJoined = path.resolve(joined);
  const resolvedRoot = path.resolve(configDir);
  if (!resolvedJoined.startsWith(resolvedRoot + path.sep) && resolvedJoined !== resolvedRoot) {
    throw new Error(`install.cjs: refusing path-traversal escape from ${configDir}: ${relPath}`);
  }
  return joined;
}

async function installRuntime(adapter, opts) {
  // Step 1: resolve config dir
  const configDir = resolveConfigDir(adapter, opts.flags || {}, process.env);
  await fsp.mkdir(configDir, { recursive: true });

  // Step 2: read priorState
  const statePath = path.join(configDir, 'oto', '.install.json');
  const priorState = readState(statePath);  // null if missing

  // Step 3: onPreInstall
  if (typeof adapter.onPreInstall === 'function') {
    adapter.onPreInstall({ configDir, priorState, otoVersion: OTO_VERSION });
  }

  // Step 4-5: copy each source dir + apply transform
  const fileEntries = [];
  for (const srcKey of SRC_KEYS) {
    const srcAbs = path.join(opts.repoRoot, adapter.sourceDirs[srcKey]);
    const dstAbs = path.join(configDir, adapter.targetSubdirs[srcKey]);
    const result = await copyTree(srcAbs, dstAbs);
    const transformFn = adapter[TRANSFORM_KEY[srcKey]];
    for (const f of result.files) {
      if (typeof transformFn === 'function') {
        const original = await fsp.readFile(f.absPath, 'utf8');
        const transformed = transformFn(original, { srcKey, relPath: f.relPath });
        if (transformed !== original) await fsp.writeFile(f.absPath, transformed);
      }
      const hex = await sha256File(f.absPath);
      fileEntries.push({ path: path.relative(configDir, f.absPath), sha256: hex });
    }
  }

  // Step 6-7: diff vs priorState — delete removed files (with path-traversal guard, T-03-06-01)
  const newPaths = new Set(fileEntries.map(f => f.path));
  for (const prior of (priorState?.files || [])) {
    if (!newPaths.has(prior.path)) {
      const safe = assertWithin(configDir, prior.path);
      await fsp.rm(safe, { force: true });
    }
  }

  // Step 8: marker injection
  const instructionPath = path.join(configDir, adapter.instructionFilename);
  const body = adapter.renderInstructionBlock({ configDir, otoVersion: OTO_VERSION });
  injectMarkerBlock(instructionPath, OPEN_MARKER, CLOSE_MARKER, body);

  // Step 9: settings merge (Phase 3: identity)
  const settingsPath = path.join(configDir, adapter.settingsFilename);
  if (fs.existsSync(settingsPath)) {
    const existing = await fsp.readFile(settingsPath, 'utf8');
    const merged = adapter.mergeSettings(existing, '');
    if (merged !== existing) await fsp.writeFile(settingsPath, merged);
  }

  // Step 10: COMMIT POINT — write state file LAST (Pitfall F)
  writeState(statePath, {
    version: 1,
    oto_version: OTO_VERSION,
    installed_at: new Date().toISOString(),
    runtime: adapter.name,
    config_dir: configDir,
    files: fileEntries,
    instruction_file: { path: adapter.instructionFilename, open_marker: OPEN_MARKER, close_marker: CLOSE_MARKER },
  });

  // Step 11: onPostInstall
  const ctx = { runtime: adapter.name, configDir, statePath, filesCopied: fileEntries.length };
  if (typeof adapter.onPostInstall === 'function') adapter.onPostInstall(ctx);
  return ctx;
}

async function uninstallRuntime(adapter, opts) { /* ... per CONTEXT.md D-10 ... */ }

async function installAll(adapters, opts) {
  // Honor opts.homeDir override (integration tests use fake home dirs)
  const homeDir = opts.homeDir || os.homedir();
  const present = detectPresentRuntimes(homeDir);
  if (present.length === 0) {
    process.stderr.write('--all: no runtimes detected (none of ~/.claude, ~/.codex, ~/.gemini exist)\n');
    const err = new Error('no runtimes detected');
    err.exitCode = 4;
    throw err;
  }
  const results = [];
  for (const runtime of present) {
    const adapter = adapters.find(a => a.name === runtime);
    if (!adapter) continue;
    results.push(await installRuntime(adapter, opts));
  }
  return results;
}

async function uninstallAll(adapters, opts) {
  const results = [];
  for (const adapter of adapters) results.push(await uninstallRuntime(adapter, opts));
  return results;
}

module.exports = { installRuntime, uninstallRuntime, installAll, uninstallAll };
