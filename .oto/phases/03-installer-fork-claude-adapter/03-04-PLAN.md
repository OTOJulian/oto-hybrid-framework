```javascript
const fsp = require('node:fs/promises');
const fs = require('node:fs');

 * relPath is relative to `dst`.
 */
async function copyTree(src, dst, opts = {}) { ... }

/**
 * Recursively remove `dst` if present. No-op if absent. force: true semantics.
 */
async function removeTree(dst) { ... }

/**
 * Walk a directory tree and return all regular file paths (relative to root).
 * Returns [] if root absent. Skips symlinks.
 */
async function walkTree(root) { ... }

/**
 * Compute SHA-256 hex digest of file bytes.
 */
async function sha256File(absPath) { ... }

module.exports = { copyTree, removeTree, sha256File, walkTree };
```

`copyTree` algorithm:
1. Check `fs.existsSync(src)` — if false, return `{ filesCopied: 0, files: [] }` (Pitfall B).
2. Check that `src` is a directory via `fs.statSync(src).isDirectory()` — if false (it's a file), throw `Error('copyTree: src must be a directory, got file: ' + src)`.
3. `await fsp.mkdir(dst, { recursive: true })`.
4. `await fsp.cp(src, dst, { recursive: true, force: true, errorOnExist: false, dereference: false })`.
   - `dereference: false` so symlinks are copied as symlinks, not their targets — but per security review (T-04-01) we want to REJECT symlinks in source instead. So before cp: walk src, throw if any entry is a symlink. Then cp with `dereference: false` is safe.
5. Walk the dst tree, return file list with `{ absPath, relPath: path.relative(dst, absPath) }`.

`sha256File`:
```javascript
async function sha256File(absPath) {
  const buf = await fsp.readFile(absPath);
  return crypto.createHash('sha256').update(buf).digest('hex');
}
```

`walkTree`:
```javascript
async function walkTree(root) {
  if (!fs.existsSync(root)) return [];
  const out = [];
  async function rec(dir) {
    const entries = await fsp.readdir(dir, { withFileTypes: true });
    for (const e of entries) {
      const p = path.join(dir, e.name);
      if (e.isDirectory()) await rec(p);
      else if (e.isFile()) out.push(p);
      // Symlinks intentionally skipped
    }
  }
  await rec(root);
  return out;
}
```

`removeTree`:
```javascript
async function removeTree(dst) {
  await fsp.rm(dst, { recursive: true, force: true });
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| repo files (`oto/commands`, `oto/agents`, etc.) → `copyTree` → user `~/.<runtime>/` | Repo files are trusted (own source); but `copyTree` is also a generic helper that could be misused with attacker-controlled `src` |
| `src` directory contents → user filesystem | Symlinks in `src` could redirect writes outside `dst` if dereferenced |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-04-01 | E (Elevation of privilege) — symlink redirect | `copyTree` writing through symlink in `src` | mitigate | (a) Pre-flight `walkTree` rejects any symlink in `src` with `Error('symlink not allowed in source tree: ...')`. (b) `fs.cp` invoked with `dereference: false` as defense in depth. Combined: no path under `dst` can ever be written outside `dst`. Specifically blocks the case where `oto/commands/evil → /etc/passwd` would otherwise have `evil`'s contents copied into the user's CLAUDE.md tree. |
| T-03-04-02 | T (Tampering) — race between mkdir and cp | `copyTree` creates `dst` with mkdir then calls fs.cp | mitigate | `fs.cp` with `force: true` overwrites without symlink-following. The brief TOCTOU window is benign for personal-use single-user filesystem (no shared `~/.claude`). |
| T-03-04-03 | I (Information disclosure) — hash leakage | `sha256File` digest in state.files[].sha256 | accept | SHA-256 of public repo files is non-secret. State file already contains relative paths only (Pitfall H mitigated in Plan 03). No PII surface. |
| T-03-04-04 | D (DoS) — large file hashing | `sha256File` reads whole file into memory | accept | At Phase 3 close, zero files are copied (D-07 scaffolding-only). Phase 4+ files are markdown (~kB each). Memory footprint negligible. Acceptable to defer streaming hash to v2 if file sizes balloon (currently no signal they will). |
| T-03-04-05 | T (Tampering) — line-ending hash drift | `sha256File` cross-platform | mitigate | Pitfall D: `.gitattributes` enforces `eol=lf` to ensure deterministic hashing across Mac dev / Linux CI. Phase 3 doesn't add `.gitattributes` (Phase 2 owns root-level config); test fixture `'hello world'` (no newline) is byte-stable. |
| T-03-04-06 | E (Elevation) — `removeTree` traversal | `fsp.rm(dst, { recursive: true, force: true })` | mitigate | `dst` is always computed by orchestrator (Plan 06) as `path.join(configDir, ...)` where `configDir` is resolved by Plan 02's `resolveConfigDir`. Path traversal in `--config-dir` is mitigated by `path.resolve()` in Plan 02 + the fact that the user supplies their own configDir — they cannot escalate their own privileges. Out-of-scope: a malicious downstream caller of `removeTree` with attacker-controlled `dst` (no such caller exists in Phase 3). |
</threat_model>
