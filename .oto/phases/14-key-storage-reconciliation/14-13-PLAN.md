<objective>
Close Gap 2 from 14-VERIFICATION.md (CR-01) and WR-07 from 14-REVIEW.md.

CR-01: `readKeyfile` in both layers returns trimmed-empty content as a present keyfile (`{ value: '', healed }`). With a valid legacy string in config and a zero-byte/whitespace keyfile on disk, `migrateLegacyIntegrationKeys` evaluates `existing.value !== current` → true, takes the keyfile-wins conflict branch (D-02), drops the only valid credential, writes `true` to config, and leaves the keyfile empty. `secretStatus` then reports `Exa: enabled — key from ~/.oto/exa_api_key ((unset))`.

WR-07: `writeFileSync`, `statSync`, `chmodSync`, and `readFileSync` all follow pre-existing symlinks — a planted `exa_api_key -> victim.txt` gets its target overwritten and re-moded to 0600 by `writeKeyfile`. Additionally, the `mode: 0o600` create option does not tighten an existing 0644 file until AFTER the new secret bytes have been written.

Fix, in one coherent readKeyfile/writeKeyfile hardening pass (single choke point both layers' migration, detection, warning, and status paths flow through):
1. lstat-based symlink/non-regular-file rejection in both readKeyfile and writeKeyfile (rejection precedes any chmod, which would dereference the link).
2. Permission healing BEFORE content read (user-mandated ordering — an empty loose-mode file still gets healed).
3. Trimmed-empty content returns null AFTER healing — an empty keyfile is not a credential, so migration overwrites it via writeKeyfile.
4. writeKeyfile heals a pre-existing loose mode to 0600 BEFORE truncation, then opens with O_WRONLY|O_CREAT|O_TRUNC|O_NOFOLLOW at mode 0600.

Purpose: SECR-03's self-heal must never destroy the only stored credential; SECR-04's status must never report an unset key as usable; SECR-01's 0600-keyfile guarantee must hold against symlink planting and loose-mode windows.
Output: hardened readKeyfile/writeKeyfile in both layers + four regression suites (CJS empty, CJS symlink, SDK empty, SDK symlink).
</objective>

    function writeKeyfile(slug, value, baseDir) {
      const base = keyfileBase(baseDir);
      const target = keyfilePath(slug, base);
      fs.mkdirSync(base, { recursive: true, mode: 0o700 });

      // OTO Phase 14 gap-closure (WR-07 / SECR-01): never write through a
      // symlink or non-regular file, and tighten a pre-existing loose mode
      // BEFORE the new secret bytes land (heal-before-truncation).
      let st = null;
      try { st = fs.lstatSync(target); } catch { /* ENOENT — fresh create below */ }
      if (st) {
        if (!st.isFile()) {
          throw new Error(`refusing to write ${target}: not a regular file — remove it and retry`);
        }
        if ((st.mode & 0o077) !== 0) fs.chmodSync(target, 0o600);
      }

      const flags = fs.constants.O_WRONLY | fs.constants.O_CREAT | fs.constants.O_TRUNC
        | (fs.constants.O_NOFOLLOW || 0);
      const fd = fs.openSync(target, flags, 0o600);
      try {
        fs.writeSync(fd, String(value) + '\n');
      } finally {
        fs.closeSync(fd);
      }
      fs.chmodSync(target, 0o600);
    }

    function readKeyfile(slug, baseDir) {
      const target = keyfilePath(slug, baseDir);

      // OTO Phase 14 gap-closure (WR-07): lstat and refuse non-regular files
      // BEFORE any chmod — chmodSync would dereference a planted symlink.
      let st;
      try {
        st = fs.lstatSync(target);
      } catch {
        return null; // ENOENT — no keyfile
      }
      if (!st.isFile()) {
        process.stderr.write(`refusing to read ${target}: not a regular file — remove it and re-set via /oto-settings-integrations\n`);
        return null;
      }

      // D-12 heal FIRST (CR-01 ordering): tighten permissions before content
      // is read, so an empty loose-mode keyfile is still healed.
      let healed = false;
      if ((st.mode & 0o077) !== 0) {
        fs.chmodSync(target, 0o600);
        healed = true;
        process.stderr.write(`fixed permissions on ${target} (now 0600)\n`);
      }

      const value = fs.readFileSync(target, 'utf8').trim();
      // OTO Phase 14 gap-closure (CR-01 / SECR-03): an empty/whitespace-only
      // keyfile is not a credential — report it absent so migration overwrites
      // it instead of treating it as an authoritative conflicting keyfile, and
      // so status reports "no key detected" instead of an enabled-unset key.
      if (value === '') return null;

      return { value, healed };
    }
