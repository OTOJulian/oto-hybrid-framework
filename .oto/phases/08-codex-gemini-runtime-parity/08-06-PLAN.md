'use strict';
const test = require('node:test');
const assert = require('node:assert/strict');
const fs = require('node:fs');
const fsp = require('node:fs/promises');
const os = require('node:os');
const path = require('node:path');
const { spawnSync } = require('node:child_process');

const REPO_ROOT = path.resolve(__dirname, '..');

function freshTmpDir(prefix) {
  return fs.mkdtempSync(path.join(os.tmpdir(), prefix));
}

function probeBinary(binName, minVersion) {
  // Returns { available: bool, version: string|null, reason: string|null }
  const r = spawnSync(binName, ['--version'], { encoding: 'utf8' });
  if (r.error || r.status !== 0) return { available: false, version: null, reason: `${binName} not on PATH` };
  const m = /(\d+\.\d+(?:\.\d+)?)/.exec(r.stdout || '');
  if (!m) return { available: false, version: null, reason: `${binName} --version unparseable: ${r.stdout}` };
  const cmp = (a, b) => a.split('.').map(Number).reduce((acc, _, i) => acc !== 0 ? acc : (Number(a.split('.')[i]||0) - Number(b.split('.')[i]||0)), 0);
  if (cmp(m[1], minVersion) < 0) return { available: false, version: m[1], reason: `${binName} v${m[1]} < required v${minVersion}` };
  return { available: true, version: m[1], reason: null };
}
