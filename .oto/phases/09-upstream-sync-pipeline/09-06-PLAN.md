       ```javascript
       function parseSyncArgs(argv) {
         const { values } = parseArgs({
           args: argv,
           options: {
             upstream:           { type: 'string' },
             to:                 { type: 'string' },
             apply:              { type: 'boolean', default: false },
             'dry-run':          { type: 'boolean', default: false },
             accept:             { type: 'string' },
             'accept-deletion':  { type: 'string' },
             'keep-deleted':     { type: 'string' },
             status:             { type: 'boolean', default: false },
             'fixture-url':      { type: 'string' },  // test-only override for upstream URL
           },
           strict: true,
           allowPositionals: false,
         });
         let mode = 'full';
         if (values.status) mode = 'status';
         else if (values.accept) mode = 'accept';
         else if (values['accept-deletion']) mode = 'accept-deletion';
         else if (values['keep-deleted']) mode = 'keep-deleted';
         // Mutual exclusion: accept-family must not combine with --upstream/--to
         if (mode !== 'full' && mode !== 'status' && (values.upstream || values.to)) {
           throw new Error('--accept / --accept-deletion / --keep-deleted cannot be combined with --upstream/--to');
         }
         if (mode === 'full' && (!values.upstream || !values.to)) {
           throw new Error('oto sync: --upstream and --to are required (or use --accept/--status)');
         }
         if (mode === 'full') validateRef(values.to);  // T-09-03
         // Default --dry-run unless --apply
         const apply = values.apply === true;
         return { mode, ...values, apply };
       }
       ```

    3. `runStage(stageName, args)`:

       ```javascript
       function runStage(stageName, args) {
         const script = STAGE_SCRIPTS[stageName];
         if (!script) throw new Error(`unknown stage: ${stageName}`);
         const r = spawnSync(process.execPath, [script, ...args], { stdio: 'inherit' });
         if (r.status === null) throw new Error(`stage ${stageName} did not run: ${r.error?.message}`);
         return r.status;
       }
       ```

       ```javascript
       async function runFullSync(parsed) {
         const upstreams = parsed.upstream === 'all' ? ['gsd', 'superpowers'] : [parsed.upstream];
         for (const u of upstreams) {
           if (u !== 'gsd' && u !== 'superpowers') {
             throw new Error(`unknown upstream '${u}' (must be gsd, superpowers, or all)`);
           }
           // 1. Pull
           const pullScript = u === 'gsd' ? 'pull-gsd' : 'pull-superpowers';
           const pullArgs = ['--to', parsed.to];
           if (parsed['fixture-url']) pullArgs.push('--url', parsed['fixture-url']);
           if (!parsed.apply) {
             // RESEARCH #2: dry-run pulls into tmpdir to avoid rotating persistent state
             const dryDir = await fsp.mkdtemp(path.join(os.tmpdir(), `oto-sync-dryrun-${u}-`));
             pullArgs.push('--dest-dir', dryDir);
             // For dry-run, override paths in subsequent stages too
             parsed._dryDir = parsed._dryDir || {};
             parsed._dryDir[u] = dryDir;
           }
           const pullStatus = runStage(pullScript, pullArgs);
           if (pullStatus !== 0) return pullStatus;
           // 2. Rebrand
           const rebrandArgs = ['--upstream', u];
           if (parsed._dryDir && parsed._dryDir[u]) {
             rebrandArgs.push('--target', path.join(parsed._dryDir[u], 'current'));
             rebrandArgs.push('--out',    path.join(parsed._dryDir[u], 'rebranded-current'));
           }
           const rebrandStatus = runStage('rebrand', rebrandArgs);
           if (rebrandStatus !== 0) return rebrandStatus;
           // 3. Merge
           const mergeArgs = ['--upstream', u];
           if (parsed.apply) mergeArgs.push('--apply');
           // W#1: dry-run pulls + rebrands into tmp dirs (parsed._dryDir[u]); merge.cjs MUST read from those
           // tmp paths via --rebranded-dir / --prior-rebranded-dir, otherwise it walks
           // .oto-sync/rebranded/<u>/current (which dry-run never wrote to → ENOENT).
           if (parsed._dryDir && parsed._dryDir[u]) {
             mergeArgs.push('--rebranded-dir', path.join(parsed._dryDir[u], 'rebranded-current'));
             mergeArgs.push('--prior-rebranded-dir', path.join(parsed._dryDir[u], 'rebranded-prior'));
             // Note: in dry-run, no `prior` snapshot exists in the tmpdir (we only pull `current`);
             // merge.cjs treats missing priorRebrandedDir as "first sync" via mergeAll's path lookups
             // which return null buffers → mergeOneFile classifies as 'added'. That's correct dry-run
             // semantics: every file looks added on first sync.
           }
           const mergeStatus = runStage('merge', mergeArgs);
           if (mergeStatus !== 0) return mergeStatus;
         }
         return 0;
       }
       ```

    5. `runStatus()`:
       ```javascript
       async function runStatus() {
         const upstreams = ['gsd', 'superpowers'];
         const lines = ['oto sync — status\n'];
         for (const u of upstreams) {
           const pinPath = path.resolve(`.oto-sync/upstream/${u}/last-synced-commit.json`);
           let pin = null;
           try { pin = JSON.parse(await fsp.readFile(pinPath, 'utf8')); } catch { /* not synced */ }
           if (pin) {
             lines.push(`  ${u}: ${pin.ref} (${pin.sha.substring(0, 7)}) at ${pin.timestamp}`);
           } else {
             lines.push(`  ${u}: not synced`);
           }
         }
         const conflictsDir = path.resolve('.oto-sync-conflicts');
         let counts = { modified: 0, added: 0, deleted: 0 };
         try {
           const walk = async (d, rel = '') => {
             for (const e of await fsp.readdir(d, { withFileTypes: true })) {
               const r = path.join(rel, e.name);
               if (e.isDirectory()) await walk(path.join(d, e.name), r);
               else if (r === 'REPORT.md') continue;
               else if (r.endsWith('.added.md')) counts.added++;
               else if (r.endsWith('.deleted.md')) counts.deleted++;
               else if (r.endsWith('.md')) counts.modified++;
             }
           };
           await walk(conflictsDir);
         } catch { /* dir may not exist yet */ }
         lines.push(`  pending conflicts: M=${counts.modified} A=${counts.added} D=${counts.deleted}`);
         process.stdout.write(lines.join('\n') + '\n');
         return 0;
       }
       ```

       ```javascript
       async function runSync(argv) {
         let parsed;
         try { parsed = parseSyncArgs(argv); }
         catch (e) { process.stderr.write(`oto sync: ${e.message}\n`); return 3; }
         try {
           if (parsed.mode === 'status') return await runStatus();
           const otoDir       = path.resolve('oto');
           const conflictsDir = path.resolve('.oto-sync-conflicts');
           const inventoryPath= path.resolve('decisions/file-inventory.json');
           const allowlistPath= path.resolve('decisions/sync-allowlist.json');
           if (parsed.mode === 'accept') {
             await acceptConflict({ relPath: parsed.accept, otoDir, conflictsDir });
             process.stdout.write(`accepted: ${parsed.accept}\n`);
             return 0;
           }
           if (parsed.mode === 'accept-deletion') {
             await acceptDeletion({ relPath: parsed['accept-deletion'], otoDir, conflictsDir, inventoryPath });
             process.stdout.write(`accepted-deletion: ${parsed['accept-deletion']}\n`);
             return 0;
           }
           if (parsed.mode === 'keep-deleted') {
             await keepDeleted({ relPath: parsed['keep-deleted'], conflictsDir, allowlistPath });
             process.stdout.write(`kept-deleted: ${parsed['keep-deleted']}\n`);
             return 0;
           }
           // full sync
           return await runFullSync(parsed);
         } catch (e) {
           process.stderr.write(`oto sync: ${e.message}\n`);
           return 1;
         }
       }
       ```
