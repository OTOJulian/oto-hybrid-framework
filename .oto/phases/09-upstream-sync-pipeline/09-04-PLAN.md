         const sidecar = assertSafeAcceptPath(conflictsDir, relPath + '.md');
         const target = assertSafeAcceptPath(otoDir, relPath);
         let raw;
         try { raw = await fsp.readFile(sidecar, 'utf8'); }

           if (e.code === 'ENOENT') throw new Error(`Conflict sidecar not found at ${sidecar}; nothing to accept`);

         const stripped = stripYamlHeader(raw);
         await fsp.mkdir(path.dirname(target), { recursive: true });
         await fsp.writeFile(target, stripped);
         await fsp.unlink(sidecar);
         return { acceptedPath: target };
       }
