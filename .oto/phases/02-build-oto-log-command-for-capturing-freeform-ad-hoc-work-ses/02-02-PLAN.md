<objective>
Implement the log-capture library `oto/bin/lib/log.cjs` that backs the `/oto-log` command. The library exposes pure helpers (slug derivation, subcommand routing), evidence capture (git diff/log/status bounded by a ref with DATA_START/DATA_END wrapping and 8KB cap), atomic file write with collision suffix, session lifecycle (start/end + .active-session.json), list/show/promote, and a CLI entrypoint dispatched from `oto-tools log` and public `oto log`. Body drafting is intentionally NOT in this library — the markdown command (Plan 03) composes the six-section body from evidence and the conversation transcript. This library returns the evidence bundle and writes whatever body bytes it is given.

    1. `function deriveLogSlug(title, { collisionSuffix } = {})`:
       - Coerce title to string, throw `Error('title required')` if falsy/empty after trim
       - words = title.toLowerCase().replace(/[^\\w\\s-]/g, ' ').split(/\\s+/).filter(Boolean)
       - while words.length and STOP_WORDS.has(words[0]): words.shift()
       - meaningful = words.slice(0, 4).join('-')
       - if !meaningful: meaningful = generateSlugInternal(title) || 'untitled'
       - if collisionSuffix > 1: return `${meaningful}-${collisionSuffix}` (helper for tests calling deriveLogSlug directly with a suffix; primary collision logic lives in writeWithCollisionSuffix)
       - return meaningful

    2. `function routeSubcommand(args)`:
       - if !args || args.length === 0: return {sub:'help', rest:[]}
       - if SUBCOMMANDS.has(args[0]): return {sub:args[0], rest:args.slice(1)}
       - return {sub:'oneshot', rest:args}

    3. `function git(gitArgs, cwd)`:
       - res = spawnSync('git', gitArgs, {cwd, encoding:'utf8'})
       - if res.status !== 0: const e = new Error(`git ${gitArgs.join(' ')}: ${(res.stderr||'').trim()}`); e.gitExitCode = res.status; throw e
       - return res.stdout

    4. `function timestampParts(date)`:
       - Accepts a Date OR a `YYYY-MM-DD HH:mm` string
       - Returns `{frontmatterDate: 'YYYY-MM-DD HH:mm', filenameStamp: 'YYYYMMDD-HHmm'}`
       - When passed a Date: pad month/day/hour/min to 2 digits, use local time

    5. `function writeWithCollisionSuffix(dir, baseName, ext, content)`:
       - fs.mkdirSync(dir, {recursive:true})
       - for attempt in 1..COLLISION_LIMIT:
         - suffix = attempt === 1 ? '' : `-${attempt}`
         - filePath = path.join(dir, `${baseName}${suffix}${ext}`)
         - try: fd = fs.openSync(filePath, 'wx'); fs.writeSync(fd, content); fs.closeSync(fd); return {path:filePath, suffix:attempt}
         - catch err: if err.code !== 'EEXIST' throw; continue
       - throw new Error(`Too many collisions for ${baseName}`)

    6. `async function captureEvidence({ since, mode, cwd })`:
       - warnings = []
       - try: HEAD = git(['rev-parse','HEAD'], cwd).trim(); catch: HEAD = null; warnings.push('git unavailable')
       - if HEAD is null: return `{diff_from:null, diff_to:'HEAD', diff_text:'', files_touched:[], status_text:'', recent_commits:'', warnings}`
       - resolvedSince = since || HEAD
       - try: rawDiff = git(['diff', `${resolvedSince}..HEAD`], cwd); catch: rawDiff = ''; warnings.push('diff failed')
       - try: filesRaw = git(['diff','--name-only',`${resolvedSince}..HEAD`], cwd); files_touched = filesRaw.split('\\n').filter(Boolean); catch: files_touched = []
       - try: status_text = git(['status','--porcelain'], cwd); catch: status_text = ''
       - try: recent_commits = git(['log','--oneline',`${resolvedSince}..HEAD`], cwd); catch: recent_commits = ''
       - cappedDiff = rawDiff.length > DIFF_CAP ? rawDiff.slice(0,DIFF_CAP) + '\\n... <truncated>' : rawDiff
       - diff_text = `<DATA_START>\\n${cappedDiff}\\n<DATA_END>`
       - wrappedCommits = recent_commits ? `<DATA_START>\\n${recent_commits}\\n<DATA_END>` : ''
       - return `{diff_from:resolvedSince, diff_to:'HEAD', diff_text, files_touched, status_text, recent_commits:wrappedCommits, warnings}`

    7. `async function writeLogEntry({ title, body, mode='oneshot', phase=null, diff_from=null, diff_to='HEAD', files_touched=[], open_questions=[], cwd, date })`:
       - if !title || !title.trim(): throw new Error('title required')
       - resolvedDate = date || new Date()
       - {frontmatterDate, filenameStamp} = timestampParts(resolvedDate)
       - slug = deriveLogSlug(title)
       - dir = path.join(planningDir(cwd), 'logs')
       - fmObject = {date:frontmatterDate, title, slug, mode, phase: phase || null, diff_from, diff_to, files_touched, open_questions, promoted:false}
       - content = normalizeMd(spliceFrontmatter(body || '', fmObject))
       - {path:filePath, suffix} = writeWithCollisionSuffix(dir, `${filenameStamp}-${slug}`, '.md', content)
       - return {path:filePath, slug, suffix, frontmatter:fmObject}

    8. `async function startSession({ title, cwd, date })`:
       - if !title || !title.trim(): throw new Error('title required')
       - resolvedDate = date || new Date()
       - sessionPath = path.join(planningDir(cwd), 'logs', '.active-session.json')
       - warnings = []
       - if fs.existsSync(sessionPath): const prior = JSON.parse(fs.readFileSync(sessionPath,'utf8')); await endSession({cwd, date:resolvedDate, autoEnded:true, priorSession:prior, body:''}); warnings.push(`auto-ended prior session: ${prior.title}`)
       - let start_ref = null
       - try: start_ref = git(['rev-parse','HEAD'], cwd).trim(); catch: warnings.push('git unavailable; start_ref=null')
       - {frontmatterDate} = timestampParts(resolvedDate)
       - sessionData = {start_ref, start_time:frontmatterDate, title}
       - fs.mkdirSync(path.dirname(sessionPath), {recursive:true})
       - atomicWriteFileSync(sessionPath, JSON.stringify(sessionData,null,2)+'\\n')
       - return {path:sessionPath, frontmatter:sessionData, warnings}

    9. `async function endSession({ closingNotes='', body, cwd, date, autoEnded=false, priorSession=null })`:
       - sessionPath = path.join(planningDir(cwd), 'logs', '.active-session.json')
       - session = priorSession || JSON.parse(fs.readFileSync(sessionPath,'utf8'))
       - resolvedDate = date || new Date()
