    function isoTimestamp() {
      return new Date().toISOString().replace(/[:.]/g, '-');
    }

    function rewriteMarkers(text) {
      return text.replace(
        /<!-- GSD:([a-z-]+-(?:start|end))( source:[^ ]+)? -->/g,
        '<!-- OTO:$1$2 -->'
      );
    }

    function rewriteFrontmatterKey(text) {
      // Only rewrite within the leading YAML frontmatter (between first two '---' lines)
      const m = text.match(/^---\n([\s\S]*?)\n---\n/);

      const fm = m[1].replace(/^gsd_state_version:/m, 'oto_state_version:');
      return text.replace(m[0], `---\n${fm}\n---\n`);

      // runtime-isolation guard (REQ-MIG-10)
      [refuse if abs is under ~/.claude or ~/.codex or ~/.gemini]

      const detection = detectGsdProject(abs);

      if (detection.conflicts.length > 0 && !opts.force) {
        const e = new Error('half-migrated state detected: ' + detection.conflicts.join(','));
        e.exitCode = 2; throw e;
      }

      // Backup
      let backupDir = null;
      if (!opts.noBackup) {
        backupDir = path.join(abs, '.oto-migrate-backup', isoTimestamp());
        const targets = listFilesToBackup(abs, opts.scope || 'planning');
        for (const rel of targets) {
          const src = safeJoin(abs, rel);
          if (!fs.existsSync(src)) continue;
          const dst = path.join(backupDir, rel);
          await fsp.mkdir(path.dirname(dst), {recursive:true});
          await fsp.copyFile(src, dst);
        }
      }

      // Stage via engine
      const stagingOut = await fsp.mkdtemp(path.join(os.tmpdir(), 'oto-migrate-out-'));
      try {
        const engineResult = await engine.run({
          mode: 'apply', target: abs, out: stagingOut, force: true,
          owner: opts.owner || 'OTOJulian', mapPath: RENAME_MAP_PATH
        });
        if (engineResult.exitCode !== 0) {
          const e = new Error('engine apply failed: exitCode=' + engineResult.exitCode);
          e.exitCode = engineResult.exitCode; throw e;
        }

        // Per-file marker + frontmatter rewrite within stagingOut
        for (const name of INSTRUCTION_FILES) {
          const p = path.join(stagingOut, name);
          if (!fs.existsSync(p)) continue;
          const text = await fsp.readFile(p, 'utf8');
          const rewritten = rewriteMarkers(text);
          if (rewritten !== text) await writeFileAtomic(p, rewritten);
        }
        const statePath = path.join(stagingOut, '.planning', 'STATE.md');
        if (fs.existsSync(statePath)) {
          const text = await fsp.readFile(statePath, 'utf8');
          const rewritten = rewriteFrontmatterKey(text);
          if (rewritten !== text) await writeFileAtomic(statePath, rewritten);
        }

        // Copy stagingOut back into abs (skip .oto-migrate-backup, .git, node_modules)
        await copyTreeAtomic(stagingOut, abs, ['.oto-migrate-backup', '.git', 'node_modules']);

    async function main(args, cwd) {
      let values;
      try {
        const parsed = require('node:util').parseArgs({
          args,
          options: {
            'dry-run': { type: 'boolean', default: false },
            apply: { type: 'boolean', default: false },
            'rename-state-dir': { type: 'boolean', default: false },
            'no-backup': { type: 'boolean', default: false },
            force: { type: 'boolean', default: false },
            scope: { type: 'string', default: 'planning' },
            owner: { type: 'string', default: 'OTOJulian' },
            'project-dir': { type: 'string' }
          },
          strict: true,
          allowPositionals: false
        });
        values = parsed.values;
      } catch (err) {
        process.stderr.write(`migrate: error: ${err.message}\n`);
        return 1;
      }

      const projectDir = path.resolve(values['project-dir'] || cwd || process.cwd());
      const mode = values.apply ? 'apply' : 'dry-run';
      const opts = {
        renameStateDir: values['rename-state-dir'],
        noBackup: values['no-backup'],
        force: values.force,
        scope: values.scope,
        owner: values.owner
      };

      try {
        const result = mode === 'apply'
          ? await apply(projectDir, opts)
          : await dryRun(projectDir, opts);
        // Non-GSD project: dry-run returns reason; CLI semantics — exit non-zero when called without flags on a non-GSD project (REQ from research)
        if (result.reason === 'no GSD signals' && !values['dry-run'] && !values.apply) {
          process.stderr.write(`migrate: not a GSD-era project: ${projectDir}\n`);
          return 1;
        }
        process.stdout.write(
          `migrate: ${mode} — files=${(result.filesChanged || result.files || []).length}, exit=${result.exitCode}\n`
        );
        return result.exitCode || 0;
      } catch (err) {
        process.stderr.write(`migrate: error: ${err.message}\n`);
        return err.exitCode || 5;
      }
    }

    module.exports = { detectGsdProject, dryRun, apply, main, rewriteMarkers, rewriteFrontmatterKey };

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| projectDir → file writes | Migrate writes inside the user's projectDir AND inside `os.tmpdir()` (for the engine staging area). Path-traversal guard (`safeJoin`) prevents writes outside projectDir. |
| Runtime config dirs | `~/.claude/`, `~/.codex/`, `~/.gemini/` are explicitly refused as projectDir (REQ-MIG-10). |
| Engine subprocess (none — same-process require) | Engine is a require()'d module; same trust domain. Engine's own `resolveOut` further restricts output to repoRoot or os.tmpdir(). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-04 | T (Tampering) — path traversal during apply | `apply()` write paths | mitigate | `safeJoin(projectDir, rel)` checks resolved path stays under projectDir realpath; throws on escape. Used for all backup-target and copy-back paths. |
| T-01-05 | I/T — data loss on failed apply | atomic write + backup | mitigate | All writes go via `writeFileAtomic` (temp file + rename); backup directory created before any mutation; staging in os.tmpdir() means engine failure never touches projectDir. |
| T-01-06 | T — marker injection / wrong-marker rewrite | `rewriteMarkers` regex | mitigate | Regex `/<!-- GSD:([a-z-]+-(?:start|end))( source:[^ ]+)? -->/g` is anchored on the documented sentinel comment shape (per oto/templates/instruction-file.md). Section name allowed character class `[a-z-]+` + start/end discriminator + optional source attribute. Won't match arbitrary user content like `<!-- GSDfoo -->` or `<!-- gsd: -->`. |
