function render(runtimeName, ctx) {
  // ctx: { otoVersion: string, repoUrl: string, ... }
  // Steps:
  //   1. Read oto/templates/instruction-file.md
  //   2. validateFencePairing(template) — throws on mismatch (Pitfall 7)

  //   4. Strip the right-runtime fence MARKERS only (keep contents)

module.exports = { render };
