    1. Top of file:
       ```javascript
       'use strict';
       const path = require('node:path');
       const fs = require('node:fs');
       const fsp = require('node:fs/promises');
       const { spawnSync } = require('node:child_process');
       ```

    2. `validateRef(ref)` — throws `Error("Invalid ref: '${ref}' contains characters outside [A-Za-z0-9._/+-]")` if ref does not match `/^[A-Za-z0-9._/+-]+$/`. This is T-09-03 mitigation; refs flow into `git clone --branch <ref>` and `git checkout <ref>` as positional arguments — even though spawnSync arg arrays prevent classic shell injection, refs containing `--upload-pack=...` or similar git-flag-shaped strings could trigger argument injection. Reject any ref shape that is not a plain ref name.

    3. `classifyRef(ref)`:
       - `if (/^[0-9a-f]{40}$/i.test(ref)) return 'sha';`
       - `if (ref === 'main' || ref === 'master') return 'branch';`
       - `return 'tag-or-branch';`

    4. `assertGitVersion()` — runs `spawnSync('git', ['--version'], { encoding: 'utf8' })`. Lenient regex `/git version (\d+)\.(\d+)/`. If no match, throw `oto sync requires git ≥ 2.0 on PATH (could not parse: '<literal stdout>')`. If major < 2, throw same error. Called once at the start of pullUpstream.

    5. `runGit(args, opts = {})` — internal helper wrapping `spawnSync('git', args, { encoding: 'utf8', ...opts })`. On `status === null` throw `git ${args.join(' ')} did not run: ${result.error?.message}`. On `status === 255` throw `git ${args.join(' ')} failed (exit 255): ${stderr.trim()}`. On `status > 0` throw `git ${args.join(' ')} failed (status ${status}): ${stderr.trim()}`. Return `result.stdout`.

    6. `resolveSha(url, ref)` — runGit(['ls-remote', url, ref]). If stdout empty, throw `git ls-remote returned no refs for '${ref}' at ${url}`. Else parse first non-empty line as `<sha>\t<refname>`, return the sha.

    7. `pathExists(p)` — `try { await fsp.access(p); return true; } catch { return false; }`.

    8. `rotateSnapshots(syncDir)`:
       ```javascript
       const prior = path.join(syncDir, 'prior');
       const current = path.join(syncDir, 'current');
       if (await pathExists(prior)) await fsp.rm(prior, { recursive: true, force: true });
       if (await pathExists(current)) await fsp.rename(current, prior);
       ```

    9. `pullUpstream({ name, url, ref, destDir })`:
       a. `assertGitVersion()`.
       b. `validateRef(ref)`.
       c. Branch warning: `if (ref === 'main' || ref === 'master') process.stderr.write(\`oto sync: warning — branch pin '${ref}' drifts silently between syncs. Prefer a tag.\n\`);`
       d. Ensure `destDir` exists: `await fsp.mkdir(destDir, { recursive: true });`
       e. Pitfall 10 short-circuit:
          ```javascript
          const pinFile = path.join(destDir, 'last-synced-commit.json');
          let resolvedSha = null;
          try { resolvedSha = resolveSha(url, ref); } catch { /* offline or non-resolvable; proceed to clone */ }
          if (resolvedSha && await pathExists(pinFile)) {
            const prior = JSON.parse(await fsp.readFile(pinFile, 'utf8'));
            const currentDir = path.join(destDir, 'current');
            if (prior.sha === resolvedSha && await pathExists(currentDir)) {
              return { ...prior, shortCircuited: true };
            }
          }
          ```
       f. `await rotateSnapshots(destDir);`
       g. Clone dispatch:
          ```javascript
          const current = path.join(destDir, 'current');
          const kind = classifyRef(ref);
          if (kind === 'sha') {
            runGit(['clone', url, current]);
            runGit(['checkout', ref], { cwd: current });
          } else {
            try {
              runGit(['clone', '--depth', '1', '--branch', ref, url, current]);
            } catch (err) {
              // Fallback: full clone + checkout (Pitfall 2 mitigation, ref might be a SHA-shaped tag)
              if (await pathExists(current)) await fsp.rm(current, { recursive: true, force: true });
              runGit(['clone', url, current]);
              runGit(['checkout', ref], { cwd: current });
            }
          }
          ```
       h. Resolve actual HEAD SHA: `const sha = runGit(['rev-parse', 'HEAD'], { cwd: current }).trim();`
       i. Build record:
          ```javascript
          const record = {
            upstream: name,
            ref_kind: kind === 'sha' ? 'sha' : (kind === 'branch' ? 'branch' : 'tag-or-branch'),
            ref,
            sha,
            timestamp: new Date().toISOString()
          };
          ```
       j. Validate against schema BEFORE write:
          ```javascript
          const result = validate(record, lastSyncedSchema);
          if (!result.valid) throw new Error(`pin record schema violation: ${result.errors.join('; ')}`);
          ```
       k. Write `await fsp.writeFile(pinFile, JSON.stringify(record, null, 2) + '\n');`
       l. Return record.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| user CLI flag → git subprocess | `--to <ref>` value flows into `git clone --branch <ref>` and `git checkout <ref>` |
| network → local snapshot | Cloned content lands in `.oto-sync/upstream/<name>/current/` (gitignored) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-03 | Tampering | `pullUpstream({ref})` → git args | mitigate | `validateRef` throws if ref does not match `/^[A-Za-z0-9._/+-]+$/`; `runGit` uses spawnSync arg array (no shell), preventing classic injection. High severity: this is the user-input → subprocess boundary. |
| T-09-02 | Information Disclosure | `--main` branch sync drift | mitigate | Pitfall 12 stderr warning; resolved SHA always recorded in `last-synced-commit.json`; user can audit drift history |
| T-09-01 | Information Disclosure | upstream URL hardcoded | accept | URLs live in pull-{gsd,superpowers}.cjs (Wave 2 plan 09-05); not user-supplied in v1; documented constraint |
| T-09-09 | Tampering | symlinks in upstream tree | accept | `git clone` respects upstream tree; rebrand engine in plan 09-05 will need to skip symlinks; not relevant to pull stage (raw clone is upstream's responsibility) |
</threat_model>
