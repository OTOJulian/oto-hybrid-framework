oto/bin/lib/log.cjs (does NOT exist yet — Wave 0 tests assert against this contract):

```javascript
// All exports are ASYNC except deriveLogSlug + routeSubcommand which are pure-sync helpers.
module.exports = {
  // SYNC. Pure helper. title:string, options:{collisionSuffix?:number} -> string slug.
  // Strips leading STOP_WORDS (a|an|the|of|to|in|for|on|at|by|with|from|and|or|but|is|was|are|were|i|we|my|our),
  // lowercases, hyphen-joins first 4 meaningful words. Falls back to core.generateSlugInternal(title) when result is empty.
  deriveLogSlug(title, options),

  // SYNC. args:string[] -> { sub: 'oneshot'|'start'|'end'|'list'|'show'|'promote'|'help', rest: string[] }
  // 'start'|'end'|'list'|'show'|'promote' match ONLY when first token. Otherwise sub='oneshot' and rest=args.
  // Empty args -> sub='help'.
  routeSubcommand(args),

  // ASYNC. Returns { diff_from, diff_to, diff_text, files_touched, status_text, recent_commits }.
  // Wraps diff_text and recent_commits in DATA_START/DATA_END markers (D-05).
  // Caps diff_text at 8 KB; appends '... <truncated>' when capped.
  // When git is unavailable: returns object with diff_from:null and diff_text:'' plus a warnings field.
  async captureEvidence({ since, mode, cwd }),

  // ASYNC. Writes one log entry under planningDir(cwd)/logs/.
  // Returns { path:string, slug:string, suffix:number, frontmatter:object }.
  // Honors collision suffix loop (D-09) using fs.openSync(path, 'wx').
  async writeLogEntry({ title, body, mode, phase, diff_from, diff_to, files_touched, open_questions, cwd, date }),

  // ASYNC. start: writes .oto/logs/.active-session.json + records start ref.
  //        Auto-ends prior active session with a warning if one exists.
  async startSession({ title, cwd, date }),

  // ASYNC. end: reads .active-session.json, captures evidence vs start ref,
  //        writes a session-mode log, deletes the json file. closing notes is optional.
  async endSession({ closingNotes, body, cwd, date }),

  // ASYNC. list: returns array of { path, frontmatter } sorted newest-first.
  async listLogs({ cwd, limit }),

  // ASYNC. show: resolves slug via suffix-match against `.oto/logs/*-{slug}.md`,
  //        returns { path, frontmatter, body, sections } where sections is parsed
  //        per-section markdown body (summary, what_changed, what_was_discussed,
  //        outcome, files_touched, open_questions).
  async showLog({ slug, cwd }),

  // ASYNC. promote: target ∈ 'quick'|'todo'. Creates downstream artifact(s),
  //        flips source log frontmatter promoted:true, returns { created:string[] }.
  //        Throws on target='plan' with message 'promote --to plan is not supported; use --to quick first'.
  async promoteLog({ slug, target, cwd }),

  // ASYNC. CLI entrypoint dispatched from oto-tools.cjs case 'log' and bin/install.js argv[0] === 'log'.
  // Parses --body, --phase, --since via node:util.parseArgs. Returns numeric exit code.
  // Empty title -> exitCode !== 0, stderr contains '/oto-log requires a title'.
  async main(args, cwd)
};
```
